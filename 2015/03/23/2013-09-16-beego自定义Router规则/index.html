beego自定义Router规则

官方的文档里各种符号被无视掉- -！害我卡了半天。。
直到后来看到作者在某个gist上帖出来的，才发现，还是在这备注下吧- -！
comments: true
featured: true

    beego.Router("/news/:all", &controllers.RController{})
    //match /news/path/to/123.html :all= path/to/123.html
    beego.Router("/user/:username([\w]+)", &controllers.RController{})
    //match /user/astaxie :username = astaxie
    beego.Router("/download/*.*", &controllers.RController{})
    //match /download/file/api.xml :path= file/api :ext=xml
    beego.Router("/download/ceshi/*", &controllers.RController{})
    //match /download/ceshi/file/api.json :splat=file/api.json
    beego.Router("/:id:int", &controllers.RController{})
    //match :id is type int ，regexp is ([0-9]+)
    beego.Router("/:hi:string", &controllers.RController{})
    //match :hi is type string。regexp is ([\w]+)
