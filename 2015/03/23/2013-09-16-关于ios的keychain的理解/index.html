关于ios的keychain的理解

先上相关代码

首先添加Security.frameworks 

    import 

    static NSString* defaultService = @"default_server";
    - (void)addSecurityItem
    {
         NSString* account = @"abellee";
         NSString* password = @"123456";
         NSData* accountData = [account dataUsingEncoding:NSUTF8StringEncoding:NSUTF8StringEncoding];
         NSData* passwordData = [password dataUsingEncoding:NSUTF8StringEncoding];

         NSMutableDictionary* dict = [[NSMutableDictionary alloc] init];
         [dict setObject:(id)kSecClassGenericPassword forKey:(id)kSecClass];
        [dict setObject:(id)kSecAttrAccessibleWhenUnlocked forKey:(id)kSecAttrAccessible];
        [dict setObject: accountData forKey:(id)kSecAttrGeneric];
        [dict setObject: accountData forKey:(id)kSecAttrAccount];
        [dict setObject: defaultService forKey:(id)kSecAttrService];
        [dict setObject: passwordData forKey:(id)kSecValueData];

        OSStatus status = SecItemAdd((CFDictionaryRef)dict, NULL);

        if(status == noError){
               //add success
        }else{
              //add failed
        }

    }

其中的kSecAttrAccount跟kSecValueData是一一对应的 就相当于帐号密码，
你也可以理解为这就是一个大字典，把kSecAttrAccount的值设置为你想定义的一个key，
把kSecValueData的值设置为key的value，
然后往同一个kSecAttrService里add这些数据，这样就可以添加多组数据了。

以下是获取的代码：

    NSMutableDictionary* dict = [[NSMutableDictionaryalloc] init];
    [dict setObject:(id)kSecClassGenericPasswordforKey:(id)kSecClass];
    [dict setObject:(id)kSecAttrAccessibleWhenUnlockedforKey:(id)kSecAttrAccessible];
    [dict setObject:identifyforKey:(id)kSecAttrService];
    [dict setObject:(id)kCFBooleanTrueforKey:(id)kSecReturnData];
    [dict setObject:(id)kCFBooleanTrueforKey:kSecReturnAttributes];
    [dict setObject:(id)kSecMatchLimitAllforKey:(id)kSecMatchLimit];

    CFArrayRef returnData = nil;

    OSStatus status = SecItemCopyMatching((CFDictionaryRef)dict, (CFTypeRef *)&returnData);
        if (status == noErr) {
            if (returnData) {
                NSArray* tempArray = (NSArray*)returnData;
                NSLog(@"%@>>>>>%@", [[NSStringalloc] initWithData:[[tempArray objectAtIndex:0] objectForKey:kSecAttrAccount] encoding:NSUTF8StringEncoding], [[NSStringalloc] initWithData:[[tempArray objectAtIndex:1] objectForKey:kSecAttrAccount] encoding:NSUTF8StringEncoding]);
            }
        }

添加了多少组键值对，最后returnData的数组里就有多少个NSDictionary。

不过搞不明白kSecUseItemList到底怎么设置，看官方的文档是说设置这个键的值为一个CFArrayRef 里面的数据都是CFDataRef 并且保持类型都一样，但没成功，哪位同学实验成功的话请指点下喔。。
